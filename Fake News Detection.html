#Preprocessing the data
#removing unwanted texts,spaces, and converting all alphabets to lower case
def wordoption(word):
    word = word.lower()
    word = re.sub('\[.*?\]', '', word)
    word = re.sub("\\W"," ",word) 
    word = re.sub('https?://\S+|www\.\S+', '', word)
    word = re.sub('<.*?>+', '', word)
    word = re.sub('[%s]' % re.escape(string.punctuation), '', word)
    word = re.sub('', '\n ', word)
    word = re.sub('\w*\d\w*', '', word)    
    return word

# stemming using SnowballStemmer 
def stemmer (word):
    word = word.split()
    words = ""
    for i in word:
            stemmer = SnowballStemmer("english")
            words += (stemmer.stem(i))+" "
    return words

def output_label(n):
    if n == 0:
        return "It's Fake News"
    elif n >= 1:
        return "It's Not A Fake News"
    
def manual_testing_func(news):
    testing_news = {"text":[news]}
    def_test = pd.DataFrame(testing_news)
    def_test["text"] = def_test["text"].apply(wordoption) 
    test = def_test["text"]
    test1 = vect.transform(test)
    LRA = LR.predict(test1)
    DTA = DT.predict(test1)
    GBCA = GBC.predict(test1)
    RFCA = RFC.predict(test1)

    return print("\n\nLR Model Prediction: {} \nDT Model Prediction: {} \nGBC Model Prediction: {} \nRFC Model Prediction: {}".format(output_label(LRA[0]),                                                                                                       output_label(DTA[0]), 
                                                                                                              output_label(GBCA[0]), 
                                                                                                              output_label(RFCA[0])))
